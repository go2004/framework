CREATE TABLE `account` (
  `id` bigint(8) NOT NULL COMMENT '帐号ID',
  `account` varchar(64) DEFAULT NULL COMMENT '平台帐号',
  `platformId` int(4) DEFAULT NULL COMMENT '渠道ID',
  `platformName` varchar(64) DEFAULT NULL COMMENT '渠道名',
  `platform_account` varchar(64) DEFAULT NULL COMMENT 'sdk账户名',
  `deviceId` varchar(64) DEFAULT NULL COMMENT '设备ID',
  `imei` varchar(64) DEFAULT NULL COMMENT '手机串号',
  `idfa` varchar(64) DEFAULT NULL COMMENT '苹果推号',
  `mac` varchar(64) DEFAULT NULL COMMENT '网卡号',
  `resVer` int(4) DEFAULT NULL COMMENT '资源版本',
  `exeVer` int(4) DEFAULT NULL COMMENT '执行程序版本',
  `gameVer` int(4) DEFAULT NULL COMMENT '游戏版本',
  `protocolVer` int(4) DEFAULT NULL COMMENT '协议版本',
  `createTime` int(4) NOT NULL DEFAULT '0' COMMENT '创建时间',
  `lastLogintime` int(4) DEFAULT NULL COMMENT '上次登录时间',
  `gsid` int(4) DEFAULT '0' COMMENT '账号第一次创建角色的GSI的',
  `lastLogoutTime` int(4) DEFAULT NULL COMMENT '上次登出时间',
  UNIQUE KEY `platAndName` (`platformId`,`account`),
  KEY `id` (`id`) USING BTREE
) ENGINE=InnoDB DEFAULT CHARSET=utf8;
